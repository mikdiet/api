def print(s)
  case s
  when /\A(.*\#.*) = \z/
    runner.new_test($1)
  when /\A(.* s) = \z/
    runner.add_status(" = "+$1.chomp)
  when /\A\.+\z/
    runner.succeed
  when /\A[EFS]\z/
    runner.failed(s)
  else
    $stdout.print(s)
  end
end
