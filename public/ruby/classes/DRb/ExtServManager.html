def regist(name, ro)
  synchronize do
    @servers[name] = ro
    @cond.signal
  end
  self
end

def service(name)
  synchronize do
    while true
      server = @servers[name]
      return server if server && server.alive? # server may be `false'
      invoke_service(name)
      @cond.wait
    end
  end
end
