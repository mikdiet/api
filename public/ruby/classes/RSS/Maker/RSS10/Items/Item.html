def to_feed(rss)
  set_default_values do
    item = RDF::Item.new(link)
    set = setup_values(item)
    if set
      item.dc_dates.clear
      rss.items << item
      set_parent(item, rss)
      setup_other_elements(rss, item)
    elsif !have_required_values?
      raise NotSetError.new("maker.item", not_set_required_variables)
    end
  end
end
