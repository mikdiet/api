def each_recursive(&block) # :yields: node
  self.elements.each {|node|
    block.call(node)
    node.each_recursive(&block)
  }
end

Find (and return) first subnode (recursively) for which the block evaluates to true. Returns nil if none was found.

Source: show | on GitHub 

# File ruby/lib/rexml/node.rb, line 63
def find_first_recursive(&block) # :yields: node
  each_recursive {|node|
    return node if block.call(node)
  }
  return nil
end

def previous_sibling_node
  return nil if @parent.nil?
  ind = @parent.index(self)
  return nil if ind == 0
  @parent[ ind - 1 ]
end

# File ruby/lib/rexml/node.rb, line 27
def to_s indent=nil
  unless indent.nil?
    Kernel.warn( "#{self.class.name}.to_s(indent) parameter is deprecated", uplevel: 1)
    f = REXML::Formatters::Pretty.new( indent )
    f.write( self, rv = "" )
  else
    f = REXML::Formatters::Default.new
    f.write( self, rv = "" )
  end
  return rv
end
