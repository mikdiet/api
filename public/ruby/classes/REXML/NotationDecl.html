# File ruby/lib/rexml/doctype.rb, line 299
def to_s
  context = nil
  context = parent.context if parent
  notation = "<!NOTATION #{@name}"
  reference_writer = ReferenceWriter.new(@middle, @public, @system, context)
  reference_writer.write(notation)
  notation << ">"
  notation
end
