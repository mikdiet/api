def initialize(size, name, func)
  @name = name
  @request_queue = Queue.new
  @response_queue = Queue.new
  @func = func
  @size = size
  @threads = nil
  SharedHelpers.trap("INT") { abort_threads }
end

def deq
  result = @response_queue.deq
  raise result.exception if result.is_a?(WrappedException)
  result
end
