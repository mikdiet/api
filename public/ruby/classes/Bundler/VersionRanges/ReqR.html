def <=>(other)
  return -1 if other.equal?(INFINITY)

  comp = left <=> other.left
  return comp unless comp.zero?

  right <=> other.right
end

def cover?(v)
  return false if left.inclusive && left.version > v
  return false if !left.inclusive && left.version >= v

  if right.version != INFINITY
    return false if right.inclusive && right.version < v
    return false if !right.inclusive && right.version <= v
  end

  true
end
