def each
  return enum_for unless block_given?
  action = @first_action
  loop do
    break unless action
    yield action
    action = action.next
  end
  self
end

# File ruby/lib/bundler/vendor/molinillo/lib/molinillo/dependency_graph/log.rb, line 63
def pop!(graph)
  return unless action = @current_action
  unless @current_action = action.previous
    @first_action = nil
  end
  action.down(graph)
  action
end

def rewind_to(graph, tag)
  loop do
    action = pop!(graph)
    raise "No tag #{tag.inspect} found" unless action
    break if action.class.action_name == :tag && action.tag == tag
  end
end
