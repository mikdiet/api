def initialize(server, name)
  super(server, name)
  @script_filename = name
  @tempdir = server[:TempDir]
  interpreter = server[:CGIInterpreter]
  if interpreter.is_a?(Array)
    @cgicmd = CGIRunnerArray + interpreter
  else
    @cgicmd = "#{CGIRunner} #{interpreter}"
  end
end
