def call(env)
  mtime = ActiveRecord::Base.connection.migration_context.last_migration.mtime.to_i
  if @last_check < mtime
    ActiveRecord::Migration.check_pending!(connection)
    @last_check = mtime
  end
  @app.call(env)
end
