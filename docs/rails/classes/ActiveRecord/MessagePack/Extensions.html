def install(registry)
  registry.register_type 119, ActiveModel::Type::Binary::Data,
    packer: :to_s,
    unpacker: :new

  registry.register_type 120, ActiveRecord::Base,
    packer: method(:write_record),
    unpacker: method(:read_record),
    recursive: true
end
