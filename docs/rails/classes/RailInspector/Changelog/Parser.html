def parse
  until @buffer.eos?
    if peek_release_header?
      pop_entry
      next parse_release_header
    end

    if peek_footer?
      pop_entry
      next parse_footer
    end

    pop_entry if peek_probably_header?

    parse_line
  end

  @entries
end
