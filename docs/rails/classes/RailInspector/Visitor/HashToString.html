def visit_assoc(node)
  @to_s << " "
  visit(node.key)

  case node.key
  when SyntaxTree::StringLiteral
    @to_s << " => "
  end

  visit(node.value)
end

def visit_hash(node)
  @to_s << "{"

  if node.assocs.length > 0
    visit(node.assocs[0])

    if node.assocs.length > 1
      node.assocs[1..-1].each do |a|
        @to_s << ","
        visit(a)
      end
    end
    @to_s << " "
  end

  @to_s << "}"
end
