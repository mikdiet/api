def bottom(n = 0)
  bind = @frames[n]
  fail FrameOverflow unless bind
  bind
end

# File ruby/lib/irb/frame.rb, line 37
def trace_func(event, file, line, id, binding)
  case event
  when 'call', 'class'
    @frames.push binding
  when 'return', 'end'
    @frames.pop
  end
end
