def initialize((h, w))
  @data = (0..h-2).map { [0] * w }
  @scale = [w / 2.0, h-2].min
  @center = Complex(w / 2, h-2)
end

def draw
  @data.each {|row| row.fill(0) }
  yield
  @data.map {|row| row.map {|n| " ',;"[n] }.join }.join("\n")
end

def line((x1, y1), (x2, y2))
  p1 = Complex(x1, y1) / 2 * @scale + @center
  p2 = Complex(x2, y2) / 2 * @scale + @center
  line0(p1, p2)
end
