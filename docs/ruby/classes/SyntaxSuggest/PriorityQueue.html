Class SyntaxSuggest::PriorityQueue < Object

  • ruby/lib/syntax_suggest/priority_queue.rb

Holds elements in a priority heap on insert

Instead of constantly calling ‘sort!`, put the element where it belongs the first time around

Example:

queue = PriorityQueue.new
queue << 33
queue << 44
queue << 1

puts queue.peek # => 44

def exchange(source, target)
  a = @elements[source]
  b = @elements[target]
  @elements[source] = b
  @elements[target] = a
end

def pop
  exchange(0, last_index)
  max = @elements.pop
  bubble_down(0)
  max
end

def sorted
  out = []
  elements = @elements.dup
  while (element = pop)
    out << element
  end
  @elements = elements
  out.reverse
end
