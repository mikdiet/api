def initialize(code_lines:, blocks:, io: $stderr, filename: nil, terminal: DEFAULT_VALUE)
  @io = io
  @blocks = Array(blocks)
  @filename = filename
  @code_lines = code_lines

  @terminal = (terminal == DEFAULT_VALUE) ? io.isatty : terminal
end

def call
  if document_ok?
    return self
  end

  if filename
    @io.puts("--> #{filename}")
    @io.puts
  end
  @blocks.each do |block|
    display_block(block)
  end

  self
end
