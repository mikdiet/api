def initialize(env)
  @home = normalize_home_dir(env["GEM_HOME"] || Gem.default_dir)
  @path = split_gem_path env["GEM_PATH"], @home

  @spec_cache_dir = env["GEM_SPEC_CACHE"] || Gem.default_spec_cache_dir
end
